cmake_minimum_required(VERSION 3.10)

project(MyTinySTL_Refactored VERSION 3.0.0 LANGUAGES CXX)

# 设置C++标准 / Set C++ Standard
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# 编译选项 / Compile Options
if(CMAKE_CXX_COMPILER_ID STREQUAL "GNU")
    if(CMAKE_CXX_COMPILER_VERSION VERSION_LESS "5.0.0")
        message(FATAL_ERROR "GCC version must be at least 5.0")
    endif()
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -Wno-sign-compare -O2")
    if(CMAKE_BUILD_TYPE STREQUAL "Debug")
        set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g -DMYSTL_DEBUG_MODE")
    endif()
elseif(CMAKE_CXX_COMPILER_ID STREQUAL "Clang")
    if(CMAKE_CXX_COMPILER_VERSION VERSION_LESS "3.5.0")
        message(FATAL_ERROR "Clang version must be at least 3.5")
    endif()
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -Wno-sign-compare -O2")
    if(CMAKE_BUILD_TYPE STREQUAL "Debug")
        set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g -DMYSTL_DEBUG_MODE")
    endif()
elseif(CMAKE_CXX_COMPILER_ID STREQUAL "MSVC")
    if(CMAKE_CXX_COMPILER_VERSION VERSION_LESS "19.0")
        message(FATAL_ERROR "MSVC version must be at least 19.0 (Visual Studio 2015)")
    endif()
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /W3 /O2")
    if(CMAKE_BUILD_TYPE STREQUAL "Debug")
        set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /DMYSTL_DEBUG_MODE")
    endif()
endif()

# 包含目录 / Include Directories
include_directories(${PROJECT_SOURCE_DIR}/MyTinySTL/include)

# 创建头文件库 / Create Header-only Library
add_library(mystl INTERFACE)
target_include_directories(mystl INTERFACE
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/MyTinySTL/include>
    $<INSTALL_INTERFACE:include>
)
target_compile_features(mystl INTERFACE cxx_std_11)

# 选项 / Options
option(MYSTL_BUILD_TESTS "Build tests" ON)
option(MYSTL_BUILD_BENCHMARKS "Build benchmarks" OFF)
option(MYSTL_BUILD_EXAMPLES "Build examples" OFF)

# 性能测试 / Benchmarks
if(MYSTL_BUILD_BENCHMARKS)

# 示例 / Examples
if(MYSTL_BUILD_EXAMPLES)
    add_subdirectory(examples)
endif()

# 安装 / Installation
install(TARGETS mystl
    EXPORT mystl-targets
    LIBRARY DESTINATION lib
    ARCHIVE DESTINATION lib
    RUNTIME DESTINATION bin
    INCLUDES DESTINATION include
)

install(DIRECTORY MyTinySTL/include/mystl
    DESTINATION include
    FILES_MATCHING PATTERN "*.h"
)

install(EXPORT mystl-targets
    FILE mystl-targets.cmake
    NAMESPACE mystl::
    DESTINATION lib/cmake/mystl
)

# 生成配置文件 / Generate Config Files
include(CMakePackageConfigHelpers)

configure_package_config_file(
    "${CMAKE_CURRENT_SOURCE_DIR}/cmake/mystl-config.cmake.in"
    "${CMAKE_CURRENT_BINARY_DIR}/mystl-config.cmake"
    INSTALL_DESTINATION lib/cmake/mystl
)

write_basic_package_version_file(
    "${CMAKE_CURRENT_BINARY_DIR}/mystl-config-version.cmake"
    VERSION ${PROJECT_VERSION}
    COMPATIBILITY SameMajorVersion
)

install(FILES
    "${CMAKE_CURRENT_BINARY_DIR}/mystl-config.cmake"
    "${CMAKE_CURRENT_BINARY_DIR}/mystl-config-version.cmake"
    DESTINATION lib/cmake/mystl
)

# 打包 / Packaging
set(CPACK_PACKAGE_NAME "MyTinySTL")
set(CPACK_PACKAGE_VERSION ${PROJECT_VERSION})
set(CPACK_PACKAGE_DESCRIPTION_SUMMARY "A lightweight STL implementation based on C++11")
set(CPACK_PACKAGE_VENDOR "MyTinySTL Contributors")
set(CPACK_PACKAGE_CONTACT "maintainer@mytinystl.org")
set(CPACK_RESOURCE_FILE_LICENSE "${CMAKE_CURRENT_SOURCE_DIR}/License.txt")
set(CPACK_RESOURCE_FILE_README "${CMAKE_CURRENT_SOURCE_DIR}/README.md")

include(CPack)

# 代码覆盖率 / Code Coverage (可选)
if(CMAKE_BUILD_TYPE STREQUAL "Debug" AND CMAKE_CXX_COMPILER_ID STREQUAL "GNU")
    option(MYSTL_ENABLE_COVERAGE "Enable code coverage" OFF)
    if(MYSTL_ENABLE_COVERAGE)
        set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} --coverage")
        set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} --coverage")
    endif()
endif()

# 静态分析 / Static Analysis (可选)
option(MYSTL_ENABLE_CLANG_TIDY "Enable clang-tidy" OFF)
if(MYSTL_ENABLE_CLANG_TIDY)
    find_program(CLANG_TIDY_EXE NAMES "clang-tidy")
    if(CLANG_TIDY_EXE)
        set(CMAKE_CXX_CLANG_TIDY "${CLANG_TIDY_EXE}")
    endif()
endif()

# 文档生成 / Documentation Generation (可选)
option(MYSTL_BUILD_DOCS "Build documentation" OFF)
if(MYSTL_BUILD_DOCS)
    find_package(Doxygen)
    if(DOXYGEN_FOUND)
        configure_file(${CMAKE_CURRENT_SOURCE_DIR}/docs/Doxyfile.in
                      ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile @ONLY)
        add_custom_target(docs
            ${DOXYGEN_EXECUTABLE} ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile
            WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
            COMMENT "Generating API documentation with Doxygen" VERBATIM
        )
    endif()
endif()

message(STATUS "  Version: ${PROJECT_VERSION}")
message(STATUS "  Build Type: ${CMAKE_BUILD_TYPE}")
message(STATUS "  C++ Standard: ${CMAKE_CXX_STANDARD}")
message(STATUS "  Compiler: ${CMAKE_CXX_COMPILER_ID} ${CMAKE_CXX_COMPILER_VERSION}")
message(STATUS "  Build Tests: ${MYSTL_BUILD_TESTS}")
message(STATUS "  Build Benchmarks: ${MYSTL_BUILD_BENCHMARKS}")
message(STATUS "  Build Examples: ${MYSTL_BUILD_EXAMPLES}")
message(STATUS "  Build Documentation: ${MYSTL_BUILD_DOCS}")
if(MYSTL_ENABLE_COVERAGE)
    message(STATUS "  Code Coverage: Enabled")
endif()
if(MYSTL_ENABLE_CLANG_TIDY)
    message(STATUS "  Static Analysis: Enabled")
endif()
